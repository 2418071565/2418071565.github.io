    int maxSubArray(vector<int>& nums) {
        function<tuple<int,int,int,int>(int,int)> dfs = [&](int l,int r)
        {
            // Âà∞Âè∂Â≠êËäÇÁÇπÂ∞±Áõ¥Êé•ËøîÂõû
            if(l == r)

            int m = (l + r) >> 1;

            auto [l_ms,l_ls,l_rs,l_sum] = dfs(l,m);

            return make_tuple(ms,ls,rs,sum);
        };
        auto [ms,ls,rs,sum] = dfs(0,nums.size() - 1);
        return ms;
    }

using namespace std;
using ll = long long;
constexpr int MAX = 2e6 + 100;
void solve();
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    solve();

struct S
{
    int l, r;       // Âå∫Èó¥Â∑¶Âè≥Á´ØÁÇπ
    ll sum;         // Âå∫Èó¥Âíå
    ll ls;          // ÂåÖÂê´Â∑¶Á´ØÁÇπÁöÑÊúÄÂ§ßÂ≠êÊÆµÂíå
    ll rs;          // ÂåÖÂê´Âè≥Á´ØÁÇπÁöÑÊúÄÂ§ßÂ≠êÊÆµÂíå

// a Ë°®Á§∫Â∑¶Âå∫Èó¥ÔºåbË°®Á§∫Âè≥Âå∫Èó¥ÔºåÂêéÈù¢‰ΩøÁî®Êó∂‰πüË¶Å‰øùËØÅËøô‰∏™È°∫Â∫è
S operator+(const S& a,const S& b)
{
    S res;
    res.sum = a.sum + b.sum;
    res.ls = max(a.ls,a.sum + b.ls);
    res.rs = max(b.rs,b.sum + a.rs);
    res.ms = max(a.rs + b.ls,max(a.ms,b.ms));
    // Áª¥Êä§Âå∫Èó¥Â∑¶Âè≥Á´ØÁÇπ
    res.l = a.l;
    res.r = b.r;
    return res;

// Ê±ÇÂ∑¶Âè≥Â≠êËäÇÁÇπÁöÑÂáΩÊï∞ÂíåÂå∫Èó¥ÊäòÂçäÁöÑÂáΩÊï∞
inline int ls(int k) { return k << 1; }
inline int rs(int k) { return k << 1 | 1; }
inline int md(int l, int r) { return (l + r) >> 1; }
int n, m;

// ÂêàÂπ∂Âå∫Èó¥
void push_up(int k)
{
    f[k] = f[ls(k)] + f[rs(k)];
}

// Âª∫Ê†ë
void build(int k, int l, int r)
{
    f[k].l = l, f[k].r = r; 
    if (l == r)
    {
        // ËæìÂÖ•Êï∞ÊçÆ
        cin >> f[k].sum;
        f[k].ls = f[k].rs = f[k].ms = f[k].sum;
        return;
    }
    int m = md(l, r); 
    build(ls(k), l, m);
    build(rs(k), m + 1, r);
    push_up(k);
}

// Êü•ËØ¢
S ask(int k, int l, int r)
{
    if (l <= f[k].l && r >= f[k].r)
        return f[k];
    int m = md(f[k].l, f[k].r);
    S res;
    if (r <= m)res = ask(ls(k), l, r);
    else if (l > m)res = ask(rs(k), l, r);
    else res = ask(ls(k), l, m) + ask(rs(k), m + 1, r);
    return res;
}

void solve()
{
    cin >> n;
    build(1,1,n);
    cin >> m;
    while(m--)
    {
        int l,r;
        cin >> l >> r;
        cout << ask(1,l,r).ms << '\n';
    }
}
