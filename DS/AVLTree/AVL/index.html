inline treenode* AVLTree::_erase(treenode* root, int data)
{
    //1.Âà†Èô§ËäÇÁÇπ
    if (root->_val < data)
        if (root->_r)
            _erase(root->_r, data);
        else if (root->_val > data)
            if (root->_l)
                _erase(root->_l, data);
            else
            {
                //Âà†Èô§Êúâ‰∏âÁßçÊÉÖÂÜµ
                //1.Âà†Èô§ËäÇÁÇπ‰∏∫Âè∂Â≠êËäÇÁÇπ 
                //Áõ¥Êé•Âà†Èô§
                if (!root->_l && !root->_r)
                {
                    if (root->_parent)
                    {
                        if (root->_parent->_l == root)
                            root->_parent->_l = nullptr;
                        else
                            root->_parent->_r = nullptr;
                    }
                    delete root;
                    root = nullptr;
                }
                //2.Âà†Èô§ËäÇÁÇπÊúâ‰∏Ä‰∏™Â≠êÊ†ë
                //Â∞ÜËØ•Â≠êÊ†ë‰ª£ÊõøÊéâÂà†Èô§ËäÇÁÇπÁöÑ‰ΩçÁΩÆ
                else if (!root->_l || !root->_r)
                {
                    if (root->_parent)
                    {
                        if (root->_parent->_l == root)
                            root->_parent->_l = root->_l ? root->_l : root->_r;
                        else
                            root->_parent->_r = root->_l ? root->_l : root->_r;
                    }
                    treenode* s = root->_l ? root->_l : root->_r;
                    s->_parent = root->_parent;
                    delete root;
                    root = s;
                }
                // 3.Âà†Èô§ËäÇÁÇπÊúâ‰∏§‰∏™Â≠êÊ†ë
                //Áî®‰∏≠Â∫èÈÅçÂéÜÁöÑÂâçÈ©±ÊàñÂêéÁªß‰ª£ÊõøËØ•ËäÇÁÇπ
                else
                {
                    treenode* lmax = get_lmax(root->_l);
                    root->_val = lmax->_val;
                    _erase(root->_l, lmax->_val);
                }
            }
    //2.Ë∞ÉÊï¥È´òÂ∫¶
    updata_deepth(root);
    //3.Ë∞ÉÊï¥ÊúÄÂ∞è‰∏çÂπ≥Ë°°Â≠êÊ†ë
    if (get_balance_factor(root) == 2)
    {
        if (get_balance_factor(root->_l) == 1)
            root = RR_roate(root);
        else
            root = LR_roate(root);
    }
    else if (get_balance_factor(root) == -2)
    {
        if (get_balance_factor(root) == -1)
            root = LL_roate(root);
        else
            root = RL_roate(root);
    }
    return root;
}
