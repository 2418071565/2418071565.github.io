void zig(int x)
{
    int ls = f[x].ls;

    {
        if (get(x, f[x].fa))f[f[x].fa].rs = ls;

    f[ls].fa = f[x].fa;
    f[x].fa = ls;
    f[x].ls = f[ls].rs;

int k_th(int x, int k)
{
    int ls = f[x].ls;
    if (f[ls].sz >= k) return k_th(ls, k);
    if (f[x].cnt + f[ls].sz < k) return k_th(f[x].rs, k - f[x].cnt - f[ls].sz);
    splay(x);
    return f[x].val;
}

void insert(int val)
{   
    // Á©∫Ê†ëÂ∞±Âª∫Ê†ë
    if (!rt)
    {
        rt = ++tot;
        f[rt].val = val;
        f[rt].sz = f[rt].cnt = 1;
        return;

            return;

                f[cur].rs = ++tot;
                f[f[cur].rs].val = val;
                f[f[cur].rs].fa = cur;

            cur = f[cur].rs;
        }
    }
}

void remove(int val)
{
    // ÂÖàÊääË¶ÅÂà†ÁöÑÁÇπÊëáÂà∞Ê†ëÊ†π
    find(val);
    if(--f[rt].cnt)return;

    int ls = f[rt].ls, rs = f[rt].rs;
    f[rt].ls = f[rt].rs = 0;

    // Â∑¶Ê†ë‰∏çÂ≠òÂú®ÔºåÁõ¥Êé•ËÆ©Âè≥Ê†ëÂΩìÊ†π„ÄÇ
    if (!ls)
    {
        rt = rs;
        f[rt].fa = 0;
        return;
    }
    f[ls].fa = 0;
    k_th(ls, f[ls].sz);
    f[rt].rs = rs;
    if (rs) f[rs].fa = rt;
    push_up(rt);
}
